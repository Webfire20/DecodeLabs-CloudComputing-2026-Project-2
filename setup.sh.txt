# Update system
sudo dnf update -y

# Start and enable Nginx

<html>
<head><title>Welcome to DecodeLabs</title></head>
<body>
  <h1>Welcome to DecodeLabs</h1>
  <p>Mission Accomplished — Server Commander Active</p>
</body>
</html>
EOF

echo "Deployment complete!"
